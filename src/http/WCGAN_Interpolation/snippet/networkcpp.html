auto generator = [](Node* X) {                                                 
  X = a.Linear(X, {4, 4, 16});                                                 
  X = a.LeakyRelu(X);                                                          
                                                                               
  X = a.Deconvolution2D(X, {5, 5}, 8, 2);                                      
  X = a.LeakyRelu(X);                                                          
                                                                               
  X = a.Deconvolution2D(X, {5, 5}, 8, 2);                                      
  X = a.LeakyRelu(X);                                                          
                                                                               
  X = a.Deconvolution2D(X, {5, 5}, 1, 1);                                      
  X = a.Sigmoid(X);                                                            
  return X;                                                                    
};                                                                             
                                                                               
auto discriminator = [](Node* X) {                                             
  X = a.Convolution2D(X, {5, 5}, 8, 1);                                        
  X = a.LeakyRelu(X);                                                          
                                                                               
  X = a.Convolution2D(X, {5, 5}, 8, 2);                                        
  X = a.LeakyRelu(X);                                                          
                                                                               
  X = a.Convolution2D(X, {5, 5}, 16, 2);                                       
  X = a.LeakyRelu(X);                                                          
                                                                               
  X = a.Linear(X, {1, 1, 1});                                                  
  return X;                                                                    
};
